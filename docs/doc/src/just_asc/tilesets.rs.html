//! Out-of-the-box tilesets for your [GridConfig](crate::GridConfig).

/// Actual ASCII characters. Nice and crunchy.
///
/// ```text
///            __d""""""""""""b__
///         _P"`                '"¶_
///      .d"`                      '"b,
///     d"                            "b
///   .P`                              '¶,
///  .P                                  ¶,
/// .P          ]b           .d`          ¶,
/// d             ¶b       .d"             b
/// [               "_   .d"               ]
/// [                 ¶bd"                 ]
/// [                .d""_                 ]
/// [              .d"    "_,              ]
/// ¶            .d"       '"b             P
/// 'b          ]"            "           d`
///  'b                                  d`
///   'b,                              .d`
///     ¶_                            _P
///      '¶_,                      ._P`
///         "b_,                ._d"
///            ""¶____________P""
/// ```
///
pub const PURE_ASCII: [char; 16] = [
    ' ',  // 0000
    '.',  // 0001
    ',',  // 0010
    '_',  // 0011
    '\'', // 0100
    ']',  // 0101
    '/',  // 0110
    'd',  // 0111
    '`',  // 1000
    '\\', // 1001
    '[',  // 1010
    'b',  // 1011
    '"',  // 1100
    '¶', //  1101
    'P',  // 1110
    '#',  // 1111
];
